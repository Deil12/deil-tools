fping工具 fping -ug ${$(ifconfig -a|grep inet|grep -v 127.0.0.1|grep -v inet6|awk '{print $2}'|tr -d "addr:")}/24 | pbcopy
-a 显示存活IP，就是在ping测结果中，过滤能ping通的IP
-u 显示不可到达IP，就是在ping测结果中，过滤不能ping通的IP
-g 指定网段，或者指定开始和结束IP
-c 指定ping次数
-b 指定icmp包大小
-f 从文件获取ping的地址，注意，不能与 -g 参数同时使用，非要结合，可以用python的循环解决
-l 无限ping
制作SSL证书 openssl genrsa -des3 -out ssl.key 1024//生成rsa私钥，des3算法，1024位强度，ssl.key是秘钥文件名。
openssl req -new -key ssl.key -out ssl.csr//生成 CSR（证书签名请求）, Common Name 应该与域名保持一致
openssl rsa -in ssl.key -out ssl_no_psw.key//终端执行删除密码命令,这里目录和生成私钥的目录一致
openssl x509 -req -days 3650 -in ssl.csr -signkey ssl_no_psw.key -out ssl.crt//生成自签名证书
制作ca证书 genrsa -out ca.key 2048
req -new -x509 -days 3650 -key ca.key -out ca.pem
genrsa -out server.key 2048
req -new -key server.key -out server.csr
x509 -req -sha256 -CA ca.pem -CAkey ca.key -CAcreateserial -days 3650 -in server.csr -out server.pem

ecparam -genkey -name secp384r1 -out client.key
req -new -key client.key -out client.csr
x509 -req -sha256 -CA ca.pem -CAkey ca.key -CAcreateserial -days 3650 -in client.csr -out client.pem

生成PKCS密钥 openssl genrsa -out pri#1.pem
openssl pkcs8 -topk8 -inform PEM -in pri#1.pem -outform PEM -nocrypt -out pri#8.pem
openssl rsa -in pri#1.pem -pubout -out public#8.pem
openssl rsa -pubin -in pub#8.pem -RSAPublicKey_out -out pub#1.pem

openssl rsa -in pub#1.pem -pubout -RSAPublicKey_in -out pub#8.pem
